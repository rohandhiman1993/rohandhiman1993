const vm = Vue.createApp({
  data() {
    return {
      firstName: "Rohan",
      lastName: "Dhiman",
      age: 0,
    };
  },
  methods: {
    fullName() {
      return `${this.firstName} ${this.lastName}`;
    },
    increament() {
      this.age++;
    },

    // v-model is built we building ourself
    updateLastName(event) {
      this.lastName = event.target.value;
    },
  },
}).mount("#app");

  },

    fullName() {

    // get argumetns first n second event
    myEvent(myname, event) {
