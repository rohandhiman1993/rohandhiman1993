tip this method remove all other fields if present inside

// only selected field replace and also insert new fields also if you create
db.posts.update({title:"post thirdupdate"}, 
{$set:{title:'first post update last one', body:'new post updated', date:Date()}}) 



// increament inbuilt increament function $inc

db.posts.update({_id: ObjectId("5e67cb2bfaa7aebeb4427a09"}, {$inc:{likes:1}})

// rename of fields likes fields name change into views

db.posts.update({_id: ObjectId("5e67cb2bfaa7aebeb4427a09")}, {$rename:{likes:'views'}})

// remove

db.posts.remove({_id: ObjectId("5e67cb2bfaa7aebeb4427a09")})

Update Row

{
  title: 'Post Two',
  body: 'New body for post 2',
  date: Date()
},
{
  upsert: true
})
Update Specific Field
db.posts.update({ title: 'Post Two' },
{
  $set: {
    body: 'Body for post 2',
    category: 'Technology'
  }
})
Increment Field ($inc)
db.posts.update({ title: 'Post Two' },
{
  $inc: {
    likes: 5
  }
})
Rename Field
db.posts.update({ title: 'Post Two' },
{
  $rename: {
    likes: 'views'
  }
})
Delete Row
db.posts.remove({ title: 'Post Four' })
Sub-Documents
db.posts.update({ title: 'Post One' },
{
  $set: {
    comments: [
      {
        body: 'Comment One',
        user: 'Mary Williams',
        date: Date()
      },
      {
        body: 'Comment Two',
        user: 'Harry White',
        date: Date()
      }
    ]
  }
})
Find By Element in Array ($elemMatch)
db.posts.find({
  comments: {
     $elemMatch: {
       user: 'Mary Williams'
       }
    }
  }
)
Add Index
db.posts.createIndex({ title: 'text' })
Text Search

    $search: "\"Post O\""
    }
})
