    IF state IS NULL THEN
        SELECT * FROM clients;
    ELSE

DELIMITER $$
CREATE PROCEDURE get_unpayment(
    client_id INT,
    OUT invoices_count INT, -- OUT MEAN PRINT IN OUTPUT
    OUT invoices_total DECIMAL(9,2) -- OUT MEAN PRINT IN OUTPUT
)
BEGIN
    SELECT COUNT(*), SUM(invoice_total)
    INTO invoices_count, invoices_total
    FROM invoices i
    WHERE i.client_id = client_id
        AND payment_total = 0;

END $$
DELIMITER ;

DELIMITER $$
CREATE PROCEDURE get_risk_factor()
BEGIN
    -- declare variables
    DECLARE risk_factor DECIMAL(9,2) DEFAULT 0;
    DECLARE invoices_total DECIMAL(9,2);
    DECLARE invoices_count INT;

    SELECT COUNT(*), SUM(invoice_total)
    INTO invoices_count, invoices_total
    FROM invoices;

    SET risk_factor = invoice_total / invoices_count * 5;

    SELECT risk_factor;

END $$
DELIMITER ;

# Trigger

A block of SQL code that automatically gets executed before or after an insert, update or delete statement.

show all triggers SHOW TRIGGERS filter trigger SHOW TRIGGERS LIKE 'payments%'

Drop trigger DROP TRIGGER IF EXISTS payements_after_insert

DELIMITER $$
CREATE TRIGGER payments_after_insert
    AFTER INSERT ON payments
    FOR EACH ROW
BEGIN
    -- right here any code
    UPDATE invoices
    SET payment_total = payment_total + --NEW, OLD
    NEW.amount
    WHERE invoice_id = NEW.invoice_id;

END $$
DELIMITTER ;

A task (or block of sql code) that gets executed according to a schedule..

List of events SHOW EVENTS LIKE 'yearly%'

Remove Event DROP EVENT IF EXISTS yearly_delete_audit_rows

-- on cron job in mysql
SET GLOBAL event_scheduler = 'ON';
SET GLOBAL event_scheduler = 'OFF';

-- create event

DELIMITER $$

CREATE EVENT yearly_delete_audit_rows
ON SCHEDULE
    -- AT '2019-01-01' once
    EVERY 1 YEAR STARTS '2019-01-01' ENDS '2029-01-01'
DO BEGIN
    DELETE FROM payements_audit
    WHERE action_date < NOW() - INTERVAL 1 YEAR;

END $$

DELIMITER ;


-- modify event

DELIMITER $$

ALTER EVENT yearly_delete_audit_rows
ON SCHEDULE
    -- AT '2019-01-01' once
    EVERY 1 YEAR STARTS '2019-01-01' ENDS '2029-01-01'
DO BEGIN
    DELETE FROM payements_audit
    WHERE action_date < NOW() - INTERVAL 1 YEAR;

END $$

DELIMITER ;

-- enalble for desable event

ALTER EVENT yearly_delete_audit_rows ENABLE/DISABLE;
