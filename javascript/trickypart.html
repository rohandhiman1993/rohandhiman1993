                # Generators and yield

                Javascript can have generator functions - functions that return a Generator that cna be lazily looped over:

                function* evens(n) {
  while (true) {
    yield n;
    n += 2;
  }
}

// make a generator
// will return even numbers 2+
let allEvents = evens(2);

// lazily get the first 10 even numbers

for (let i = 0; i < 10; i += 1) {
  console.log(allEvents.next().value); //2 4 6 8 10 12 14
}

function* myCats() {
  yield "blue";
  yield "yellow";
  yield "green";
  yield "orange";
}

const catGenerator = myCats();

//  generator is like object
console.log(catGenerator.next()); // { value: 'blue', done: false }
console.log(catGenerator.next().value);
console.log(catGenerator.next().value);
console.log(catGenerator.next().value);

// fib with generator

function* fib() {
  let a = 0,
    b = 1;

  while (true) {
    yield a;
    [a, b] = [b, a + b];
  }
}

const fibGen = fib();
console.log(fibGen.next()); //0
console.log(fibGen.next()); // 1
console.log(fibGen.next()); // 1
console.log(fibGen.next()); // 2
console.log(fibGen.next()); // 3

                Best use generator example

                const allImages = Array.from(
  { length: 1000 },
  (_, i) => `https://placeimg.com/640/480/any?image=${i}`
);

function* getBatchOfImages(images, batchSize = 10) {
  let curIndex = 0;
  while (curIndex < images.length) {
    yield images.slice(curIndex, curIndex + batchSize);
    curIndex += batchSize;
  }
}

const imageGen = getBatchOfImages(allImages);
console.log(imageGen.next());
console.log(imageGen.next());
