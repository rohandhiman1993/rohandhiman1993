                    # bind with timers

                    class Counter {
  counterId;
  constructor(startingNum = 0, incrementAmt = 1) {
    this.count = startingNum;
    this.incrementAmt = incrementAmt;
  }

  // output NaN // we are getting window object inside setInterval
  // because window.setInterval live inside window object
  //   start() {
  //     setInterval(function () {
  //       console.log(this); // we are getting window object inside
  //       console.log(this.count);
  //       this.count += this.incrementAmt;
  //     }, 1000);
  //   }

  // solution

  start() {
    this.counterId = setInterval(
      function () {
        // console.log(this);
        console.log(this.count);
        this.count += this.incrementAmt;
      }.bind(this), // we bind here this our
      1000
    );

    this.stop();
  }

  stop() {
    setTimeout(() => {
      clearInterval(this.counterId);
      console.log("Interval stopped after 5 seconds.");
    }, 5000);
  }
}

const counter = new Counter(1, 10);
counter.start();

                    # Arrow function this

                    Arrow functions don't make their own this.

                    class Counter {
  counterId;
  constructor(startingNum = 0, incrementAmt = 1) {
    this.count = startingNum;
    this.incrementAmt = incrementAmt;
  }

  // output NaN // we are getting window object inside setInterval
  // because window.setInterval live inside window object
  //   start() {
  //     setInterval(function () {
  //       console.log(this); // we are getting window object inside
  //       console.log(this.count);
  //       this.count += this.incrementAmt;
  //     }, 1000);
  //   }

  // solution

  start() {
    this.counterId = setInterval(
      () => {
        // work fine because setInterval inside arrow function and arrow function always take parent (this) keyword
        console.log(this.count);
        this.count += this.incrementAmt;
      }, // we bind here this our
      1000
    );

    this.stop();
  }

  stop() {
    setTimeout(() => {
      clearInterval(this.counterId);
      console.log("Interval stopped after 5 seconds.");
    }, 5000);
  }
}

const counter = new Counter(1, 10);
counter.start();
