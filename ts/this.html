# Inheritance only Property not methods

function Child(name, email, peopleReporting) { // this is inheritance Parent.apply(this, [name, email]); this.peopleReporting = peopleReporting; }

function Parent(name, email) {
  this.name = name;
  this.email = email;
  this.points = 0;
}

Parent.prototype.login = function () {
  console.log(this.name, "Has logged in");
};

Parent.prototype.logout = function () {
  console.log(this.name, "Has logged out");
};

Parent.prototype.addPoint = function () {
  this.points++;
  console.log("total points", this.points);
};

/////// inheritance parent class
function Child(name, email, peopleReporting) {
  // this is inheritance
  Parent.apply(this, [name, email]);
  this.peopleReporting = peopleReporting;
}

const parent = new Parent("John", "john@email.com"); // Parent { name: 'John', email: 'john@email.com', points: 0 }

const child = new Child("Mark", "mark@email.com", 10);
/* 
Child {
  name: 'Mark',
  email: 'mark@email.com',
  points: 0,
  peopleReporting: 10
}
*/

console.log(parent);
console.log(child);

# Inheritance methods

Child.prototype = Object.create(Parent.prototype);

function Parent(name, email) {
  this.name = name;
  this.email = email;
  this.points = 0;
}

Parent.prototype.login = function () {
  console.log(this.name, "Has logged in");
};

Parent.prototype.logout = function () {
  console.log(this.name, "Has logged out");
};

Parent.prototype.addPoint = function () {
  this.points++;
  console.log("total points", this.points);
};

/////// inheritance parent class properties
function Child(name, email, peopleReporting) {
  // this is inheritance properties
  Parent.apply(this, [name, email]);
  this.peopleReporting = peopleReporting;
}

// /////////// this is method inheritance    /////////////
Child.prototype = Object.create(Parent.prototype);

/////// add method inside child class /////////
Child.prototype.updatePeopleReporting = function (newNumber) {
  this.peopleReporting = newNumber;
};

const parent = new Parent("John", "john@email.com"); // Parent { name: 'John', email: 'john@email.com', points: 0 }

const child = new Child("Mark", "mark@email.com", 10);
/* 
Child {
  name: 'Mark',
  email: 'mark@email.com',
  points: 0,
  peopleReporting: 10
}
*/

console.log(parent);
console.log(child);
console.log(child.login()); // Mark Has logged in

# Object

// value -> The value of the property
// writable (boolean) -> whether this property in question is writable or not
// enumerable (boolean)  -> whether you can enumerate or loop through this property or not
// configurable (boolean) -> The configurable property tells whether the user has permission
//								           to change property descriptor such as to change the value of
//                           writable and enumerable settings.

const book = {
  title: "the title",
  pages: 300,
  author: "John",
};

console.log(Object.getOwnPropertyDescriptors(book));

# read and write propety can stop

const book = new Object();

Object.defineProperty(book, "title", {
  value: "This is the title of the book",
  writeable: true,
  enumerable: true,
  configurable: true,
});

Object.defineProperty(book, "author", {
  value: "John",
  writeable: false,
  enumerable: true,
  configurable: true,
});

book.author = "Mark";

console.log(book);
