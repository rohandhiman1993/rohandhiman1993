We look at what are polymorphic functions, why we need them and how we can create them in Ts using Function overloading

type Person = {
  name: string,
  age: number,
};

function intro(name: string, age: number): Person {
  return { name, age };
}

function intro2(name: string, age: number): string {
  return name + age;
}

function intro3(name: string, age: number): void {
  console.log(name, age);
}

const intro4 = function (name: string, age: number): void {
  console.log(name, age);
};

const intro5 = (name: string, age: number): void => {
  console.log(name, age);
};

# Default and Optional parameters

// default and optional value pass
function intro(name: string, age: number = 20, country?: string) {
  return `My name is ${name} and age is ${age} my country ${country}`;
}

console.log(intro("rohan")); // My name is rohan and age is 20 my country undefined

# custom parameters and return types

enum AgeUnit {
  years = "years",
  month = "months",
}
type Person = {
  name: string;
  age: number;
  country: string;
  ageUnit: AgeUnit;
};

const person: Person = {
  name: "rohan",
  age: 30,
  ageUnit: AgeUnit.years,
  country: "usa",
};

function convertAgeToMonths(person: Person): Person {
  person.age = person.age * 12;
  person.ageUnit = AgeUnit.month;
  return person;
}

console.log(convertAgeToMonths(person));
/*
{ name: 'rohan',
age: 276,
 ageUnit: 'months',
  country: 'usa' }

*/

# Call signatures

enum AgeUnit {
  years = "years",
  month = "months",
}

// this  is call signatures
type greetingFunction = (greeting: string) => string;

type Person = {
  name: string;
  age: number;
  country: string;
  ageUnit: AgeUnit;
  // greet: Function; // this is function only check
  // replace with call signatures
  // greet: (greeting: string, age: number, ageUnit: AgeUnit) => string;
  greet: greetingFunction;
};

const person: Person = {
  name: "rohan",
  age: 30,
  ageUnit: AgeUnit.years,
  country: "usa",
  greet: (greeting: string) => {
    return `${greeting} ${person.name}`;
  },
};

function convertAgeToMonths(person: Person): Person {
  person.age = person.age * 12;
  person.ageUnit = AgeUnit.month;
  return person;
}

console.log(convertAgeToMonths(person));
/*
{
  name: 'rohan',
  age: 360,
  ageUnit: 'months',
  country: 'usa',
  greet: [Function: greet]
}

*/

console.log(person.greet("hello")); //hello rohan

# function overloading

you can have multiple functions

1 with the same name

2 but different parameter types and return type.

3 however, the number of parameters should be the same.

// parameters should be same
function sum(a: string, b: string): string;
function sum(a: number, b: number): number;

function sum(a: any, b: any): any {
  return a + b;
}

console.log(sum("1", "3")); // 13
console.log(sum(1, 3)); // 4

Advance

type Reservation = {
  departureDate: Date;
  returnDate?: Date;
  departingFrom: string;
  destination: string;
};

type Reserve = {
  (
    departureDate: Date,
    returnDate: Date,
    departingFrom: string,
    destination: string
  ): Reservation | never;
  (departureDate: Date, departingFrom: string, destination: string):
    | Reservation
    | never;
};

const reserve: Reserve = (
  departureDate: Date,
  returnDateOrDepartingFrom: Date | string,
  departingFromOrDestination: string,
  destination?: string
) => {
  if (returnDateOrDepartingFrom instanceof Date && destination) {
    return {
      departureDate: departureDate,
      returnDate: returnDateOrDepartingFrom,
      departingFrom: departingFromOrDestination,
      destination: destination,
    };
  } else if (typeof returnDateOrDepartingFrom === "string") {
    return {
      departureDate: departureDate,
      departingFrom: returnDateOrDepartingFrom,
      destination: departingFromOrDestination,
    };
  }
  throw new Error("Please provide valid details to reserve a ticket");
};

console.log(reserve(new Date(), new Date(), "New York", "Washington"));
console.log(reserve(new Date(), "New York", "Washington"));
