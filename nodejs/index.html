//  first middleware will check have value or not
app.use((req, res, next) => {
    // check check session visits page not create this variable

        req.session.visits = {
            '/':0,

        }

// second middleware checking ip address and storing in session
app.use((req,res, next) => {

    next();

# Login & singup with session

Install this packages npm i express express-session bcrypt body-parser mongodb

# package.json

{
  "name": "authentication",
  "version": "1.0.0",
  "description": "authentication",
  "main": "server.js",
  "scripts": {
    "test": "echo \"Error: no test specified\" && exit 1",
    "start":"nodemon server.js"
  },
  "author": "rohan dhiman",
  "license": "ISC",
  "dependencies": {
    "bcrypt": "^5.0.0",
    "body-parser": "^1.19.0",
    "express": "^4.17.1",
    "express-session": "^1.17.1",
    "mongodb": "^3.6.2"
  }
}

# server.js

const { request } = require("express");
const express = require("express");
const session = require("express-session");
const bodyParser = require("body-parser");
const bcrypt = require("bcrypt");
const app = express();

const mongodb = require("mongodb");
const path = require("path");

// reading package json file
const package = require(path.join(__dirname,  'package.json'));

// init body parser use for form data
app.use(bodyParser.urlencoded({extended:true}));

// init mongodb 
mongodb.connect('mongodb://127.0.0.1:27017',(error, client) => {
    if(error) return console.error(error);
    let db = client.db(package.name)

    // session init here 
    app.use(session({
        secret:'324234-23423-32423-25-435-34635',
        resave:false,
        saveUninitialized:true
    }));


    // checking here user is login or not we will use app.get, post both
    const authorize = (req, res, next) => {
        if(!req.session.user){
            next(new Error("Not logged in"))
        }else next()
    }
   
    app.get('/', (req, res, next)=> {
        if(req.session.user) return res.redirect('/private');
        res.set('Content-Type', 'text/html');
        res.send(
            `<h3>Log in</h3>
            <form action="/login" method="post">
            <input type="text" name="username" />
            <input type="password" name="password" />
            <button type="submit">Login in </button>

            </form>

            <h3>Sign in</h3>
            <form action="/signup" method="post">
            <input type="text" name="username" />
            <input type="password" name="password" />
            <button type="submit">Sign up </button>
            
            </form>
        `)
    })

    // 
    app.post('/login',(req, res, next) => {
        if(!req.body.password || !req.body.username) return next(new Error("Must provide username and password"));
        db.collection("users").findOne({username:req.body.username}, (error, user) => {

        if (error) return next(error)
      if (!user) return next(new Error('User name and/or password is wronguu')) // It's username but do not tell which one for security reasons
      bcrypt.compare(req.body.password, user.password, (error, matched) => {
        if (!error && matched) {
          req.session.user = {username: user.username}
          res.redirect('/private')
        } else next(new Error('User name and/or password is wrongpp')) // It's password but do not tell which one for security reasons
      })
    })
    })

    // destroying session
    app.get('/logout',(req, res, next) => {
        req.session.destroy();
        res.redirect("/");
    })

    app.post('/signup', (req, res, next) => {
        if (!req.body.password || !req.body.username) return next(new Error('Must provide username and password'))
        bcrypt.hash(req.body.password, 10, (error, hash) => { // 10 is good enough but you can make it more secure with 11
          if (error) return next(error)
          db.collection('users').insert({username: req.body.username, password: hash}, (error, results) => {
            if (error) return next(error)
            req.session.user = {username: results.ops[0].username}
            res.redirect('/private')
          })
        })
      })


    // we are usting here authorize method for private  
    app.get('/private', authorize, (req, res, next) => {
        res.send(`i ${req.session.user.username} <a href="/logout">Logout</a>`);
    })


    // Getting all error here
    app.use((error, req, res, next) => {
        console.log(error)
        res.send(error.message)
    })


app.listen(3000)

})

# server.js

const express = require('express')
const session = require('express-session')
const app = express()
const MongoStore = require('connect-mongo')(session)

// store session database with store mongodb
app.use(session({
  secret: '9C799922-B33B-4438-9269-B7607E9111D7',
  resave: false,
  saveUninitialized: true,
  store: new MongoStore({ url: 'mongodb://localhost:27017/web-sessions'})
}))

app.use((req, res, next) => {
  if (!req.session.visits) {
    req.session.visits = {
      '/': 0,
      '/private': 0
    }
  }
  next()
})

app.use((req, res, next) => {
  const ip = req.headers['x-forwarded-for'] || req.connection.remoteAddress
  req.session.ip = ip
  next()
})

app.get('/', (req, res, next) => {
  ++req.session.visits['/']
  res.send(`Your visits  ${JSON.stringify(req.session.visits)}  times. Last time from ${req.session.ip}`)
})

app.get('/private', (req, res, next) => {
  ++req.session.visits['/private']
  res.send(`Your visits  ${JSON.stringify(req.session.visits)}  times. Last time from ${req.session.ip}`)
})

// Crash program
setTimeout(() => { // Mocks a memory leak, restart the server and your visits will still be there!
  throw Error
}, 5000)

app.listen(3000)

# server.js

const csrf = require('csurf')
const express = require('express')
var cookieParser = require('cookie-parser')
const cors = require('cors')

const app = express()
app.use(cors())
app.use(cookieParser())
const csrfProtection = csrf({ cookie: true })

app.get('/transaction', csrfProtection, (req, res) => {
  const form = `
    <script>
    const token= "${req.csrfToken()}"
    console.log(token)
    function sendData() {
      const data = {
        amount: document.getElementsByName('amount')[0].value,
        to: document.getElementsByName('to')[0].value,
        from: document.getElementsByName('from')[0].value
      }
      console.log(data)
      fetch('/process-transaction', {
        body: JSON.stringify(data),
        credentials:'include',
        method: 'POST',
        headers: new Headers({
          'Content-Type': 'application/json',
          'CSRF-Token': token
        })}).then(res => res.json())
      .catch(error => console.error('Error:', error))
      .then(response => console.log('Success:', response));
      }
    </script>
    <form action="/csrf-process-transaction" method="POST">

      Amount: <input type="text" name="amount" value="100"  />
      To: <input type="text" name="to" value="Azat Mardan" />
      From: <input type="text" name="from" value="Johnny Pineappleseed"/>
      <button type="button" onClick="sendData()">Submit</button>
    </form>`
  res.set('Content-Type', 'text/html')
  res.send(form)
})

app.post('/process-transaction', csrfProtection, (req, res) => {
  res.send({msg: 'data is being processed'})
})

app.post('/no-csrf-process-transaction', (req, res) => {
  res.send({ msg: 'data is being processed' })
})

app.all('*', (req, res) => {
  res.send('did you mean to go to /transaction?<br/><a href="/transaction">yes</a>')
})

app.listen(3000)
